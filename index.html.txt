function makeDeck() {
  const deck = [];
  for (const s of suits) {
    for (const r of ranks) {
      deck.push({ rank:r, suit:s, name:r+s });
    }
  }
  return deck;
}

function setMode(newMode) {
  mode = newMode;

function drawDeck() {

function parseCard(name) {
  const suit = name.slice(-1);
  const rank = name.slice(0, -1);
  return { rank, suit, name };
}

function shuffle(array) {
  const arr = [...array];
  for (let i = arr.length - 1; i > 0; i--) {
    const j = Math.floor(Math.random() * (i + 1));
    [arr[i], arr[j]] = [arr[j], arr[i]];
  }
  return arr;
}

function combinations5(cards) {
  const result = [];
  for (let a = 0; a < cards.length - 4; a++)
  for (let b = a + 1; b < cards.length - 3; b++)
  for (let c = b + 1; c < cards.length - 2; c++)
  for (let d = c + 1; d < cards.length - 1; d++)
  for (let e = d + 1; e < cards.length; e++) {
    result.push([cards[a], cards[b], cards[c], cards[d], cards[e]]);
  }
  return result;
}

function evaluate5(cards) {
  const values = cards.map(c => rankValue[c.rank]).sort((a,b) => b-a);
  const flush = cards.every(c => c.suit === cards[0].suit);

  let unique = [...new Set(values)].sort((a,b) => b-a);
  let straightHigh = null;

  if (unique.length === 5) {
    if (unique[0] - unique[4] === 4) straightHigh = unique[0];
    else if (JSON.stringify(unique) === JSON.stringify([14,5,4,3,2])) straightHigh = 5;
  }

  const counts = {};
  for (const v of values) counts[v] = (counts[v] || 0) + 1;

  const groups = Object.entries(counts)
    .map(([v,c]) => ({ value:Number(v), count:c }))
    .sort((a,b) => b.count - a.count || b.value - a.value);

  if (flush && straightHigh) return [8, straightHigh];
  if (groups[0].count === 4) return [7, groups[0].value, groups[1].value];
  if (groups[0].count === 3 && groups[1].count === 2) return [6, groups[0].value, groups[1].value];
  if (flush) return [5, ...values];
  if (straightHigh) return [4, straightHigh];
  if (groups[0].count === 3) return [3, groups[0].value, ...groups.slice(1).map(g => g.value).sort((a,b) => b-a)];
  if (groups[0].count === 2 && groups[1].count === 2) {
    return [2, groups[0].value, groups[1].value, groups[2].value];
  }
  if (groups[0].count === 2) {
    return [1, groups[0].value, ...groups.slice(1).map(g => g.value).sort((a,b) => b-a)];
  }
  return [0, ...values];
}

function evaluate7(cards) {
  const hands = combinations5(cards);
  let best = evaluate5(hands[0]);

  for (const h of hands) {
    const score = evaluate5(h);
    if (compareScores(score, best) > 0) best = score;
  }

  return best;
}

function compareScores(a, b) {
  for (let i = 0; i < Math.max(a.length, b.length); i++) {
    const x = a[i] || 0;
    const y = b[i] || 0;
    if (x > y) return 1;
    if (x < y) return -1;
  }
  return 0;
}

function combination(n, r) {
  if (r < 0 || r > n) return 0;
  let result = 1;
  for (let i = 1; i <= r; i++) {
    result *= (n - i + 1) / i;
  }
  return result;
}

function hitProbability(deckSize, targetCount, drawCount) {
  if (drawCount <= 0 || targetCount <= 0) return 0;
  const miss = combination(deckSize - targetCount, drawCount) / combination(deckSize, drawCount);
  return 1 - miss;
}
